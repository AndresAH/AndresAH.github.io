/* make a canvas with https://cdn.pixabay.com/photo/2017/08/30/01/05/milky-way-2695569_960_720.jpg as the background */ var canvas = document.createElement('canvas'); canvas.width = window.innerWidth; canvas.height = window.innerHeight; canvas.style.backgroundImage = 'url(https://cdn.pixabay.com/photo/2017/08/30/01/05/milky-way-2695569_960_720.jpg)'; document.body.appendChild(canvas); /* make the background fit screen */ canvas.style.backgroundSize = 'cover'; /* draw stars going right to left. */ var ctx = canvas.getContext('2d'); var stars = []; for (var i = 0; i < 100; i++) { stars.push({ x: Math.random() * canvas.width, y: Math.random() * canvas.height, radius: Math.random() * 2 + 1, speed: Math.random() * 0.1 + 0.05 }); } function draw() { ctx.clearRect(0, 0, canvas.width, canvas.height); for (var i = 0; i < stars.length; i++) { var star = stars[i]; ctx.beginPath(); ctx.arc(star.x, star.y, star.radius, 0, Math.PI * 2, false); ctx.fill(); star.x -= star.speed; if (star.x < 0) { star.x = canvas.width; star.y = Math.random() * canvas.height; } } requestAnimationFrame(draw); } draw(); /* make starts white */ ctx.fillStyle = 'white'; /* make stars glow */ ctx.shadowColor = 'white'; ctx.shadowBlur = 10; /* make stars move 5x faster */ for (var i = 0; i < stars.length; i++) { stars[i].speed *= 5; } /* play music file in loop : https://www.soundhelix.com/examples/mp3/SoundHelix-Song-1.mp3 */ var audio = document.createElement('audio'); audio.src = 'https://www.soundhelix.com/examples/mp3/SoundHelix-Song-1.mp3'; audio.loop = true; audio.play();
